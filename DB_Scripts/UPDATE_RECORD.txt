CREATE DEFINER=`root`@`localhost` PROCEDURE `UPDATE_RECORD`(in p_PART_HISTORY_ID int,
in p_CUSTOMER_ID int,
in p_OPERATOR_ID int,  
in p_PART_NUMBER VARCHAR (128),
in p_DATE_DUE DATE,
in p_PURCHASE_ORDER_NUMBER VARCHAR (128),
in p_QTY int,
in p_OPERATIONS VARCHAR (128), 
in p_DATE_RECEIVED DATE,
in p_TO_DELETE tinyint)
BEGIN
UPDATE PART_HISTORY
SET 
 CUSTOMER_ID = P_CUSTOMER_ID,
 OPERATOR_ID = p_OPERATOR_ID,
 PART_NUMBER = p_PART_NUMBER,
 DATE_DUE = p_DATE_DUE,
 PURCHASE_ORDER_NUMBER = p_PURCHASE_ORDER_NUMBER,
 QTY = p_QTY,
 OPERATIONS = p_OPERATIONS,
 DATE_RECEIVED = p_DATE_RECEIVED,
TO_DELETE = p_TO_DELETE
WHERE  PART_HISTORY_ID = P_PART_HISTORY_ID;
END
