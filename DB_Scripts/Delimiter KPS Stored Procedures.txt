CREATE DEFINER=`root`@`localhost` PROCEDURE `DATE_DUE_RANGE_QUERY`(IN dateB text, In dateE text, In pageSize int, In offsetNum int)
BEGIN
SELECT PART_HISTORY.PART_HISTORY_ID
	,CUSTOMERS.CUSTOMER_NAME
	,OPERATORS.OPERATOR_NAME
    ,PART_HISTORY.PART_NUMBER
    ,PART_HISTORY.QTY
    ,PART_HISTORY.DATE_RECEIVED
    ,PART_HISTORY.DATE_DUE
    ,PART_HISTORY.OPERATIONS
    ,PART_HISTORY.PURCHASE_ORDER_NUMBER
FROM ((CUSTOMERS
	INNER JOIN PART_HISTORY ON CUSTOMERS.CUSTOMER_ID = PART_HISTORY.CUSTOMER_ID)
    INNER JOIN OPERATORS ON PART_HISTORY.OPERATOR_ID = OPERATORS.OPERATOR_ID)
    INNER JOIN JOBS ON OPERATORS.JOB_ID = JOBS.JOB_ID
WHERE (((PART_HISTORY.DATE_DUE) between dateB AND dateE))
AND (PART_HISTORY.TO_DELETE = FALSE)
ORDER BY PART_HISTORY.DATE_DUE
LIMIT pageSize OFFSET offsetNum;
END //

CREATE DEFINER=`root`@`localhost` PROCEDURE `CREATE_RECORD`(
in PART_HISTORY_ID int,
in CUSTOMER_ID int,
in OPERATOR_ID int,  
in PART_NUMBER VARCHAR (128),
in DATE_DUE DATE,
in PURCHASE_ORDER_NUMBER VARCHAR (128),
in QTY int,
in OPERATIONS VARCHAR (128), 
in DATE_RECEIVED DATE,
in TO_DELETE tinyint)
BEGIN INSERT INTO PART_HISTORY (PART_HISTORY_ID, CUSTOMER_ID, OPERATOR_ID, PART_NUMBER, DATE_DUE, PURCHASE_ORDER_NUMBER, QTY, OPERATIONS, DATE_RECEIVED ,TO_DELETE)
VALUES(PART_HISTORY_ID, CUSTOMER_ID, OPERATOR_ID, PART_NUMBER, DATE_DUE, PURCHASE_ORDER_NUMBER, QTY, OPERATIONS, DATE_RECEIVED , TO_DELETE);

END//

CREATE DEFINER=`root`@`localhost` PROCEDURE `GET_CUSTOMERS`()

CREATE DEFINER=`root`@`localhost` PROCEDURE `UPDATE_RECORD`(in p_PART_HISTORY_ID int,
in p_CUSTOMER_ID int,
in p_OPERATOR_ID int,  
in p_PART_NUMBER VARCHAR (128),
in p_DATE_DUE DATE,
in p_PURCHASE_ORDER_NUMBER VARCHAR (128),
in p_QTY int,
in p_OPERATIONS VARCHAR (128), 
in p_DATE_RECEIVED DATE,
in p_TO_DELETE tinyint)
BEGIN
UPDATE PART_HISTORY
SET 
 CUSTOMER_ID = P_CUSTOMER_ID,
 OPERATOR_ID = p_OPERATOR_ID,
 PART_NUMBER = p_PART_NUMBER,
 DATE_DUE = p_DATE_DUE,
 PURCHASE_ORDER_NUMBER = p_PURCHASE_ORDER_NUMBER,
 QTY = p_QTY,
 OPERATIONS = p_OPERATIONS,
 DATE_RECEIVED = p_DATE_RECEIVED,
TO_DELETE = p_TO_DELETE
WHERE  PART_HISTORY_ID = P_PART_HISTORY_ID;
END//


CREATE DEFINER=`root`@`localhost` PROCEDURE `USER_NAME_QUERY`(IN userName text)
BEGIN
SELECT
	USERS.USER_ID,
	USERS.USER_NAME,
    ROLES.ROLES_ID,
    ROLES.ROLE_NAME
FROM USERS
JOIN USER_ROLES on USERS.USER_ID = USER_ROLES.USER_ID
JOIN ROLES on ROLES.ROLES_ID = USER_ROLES.ROLE_ID
WHERE USERS.USER_NAME = username;
END//

CREATE DEFINER=`root`@`localhost` PROCEDURE `USER_NAME_QUERY_COUNT`(IN userName text)
BEGIN
SELECT COUNT(USER_ID)
FROM USERS
WHERE USERS.USER_NAME = userName;
END//
