CREATE DEFINER=`root`@`localhost` PROCEDURE `CREATE_RECORD`(
in PART_HISTORY_ID int,
in CUSTOMER_ID int,
in OPERATOR_ID int,  
in PART_NUMBER VARCHAR (128),
in DATE_DUE DATE,
in PURCHASE_ORDER_NUMBER VARCHAR (128),
in QTY int,
in OPERATIONS VARCHAR (128), 
in DATE_RECEIVED DATE,
in TO_DELETE tinyint)
BEGIN INSERT INTO PART_HISTORY (PART_HISTORY_ID, CUSTOMER_ID, OPERATOR_ID, PART_NUMBER, DATE_DUE, PURCHASE_ORDER_NUMBER, QTY, OPERATIONS, DATE_RECEIVED ,TO_DELETE)
VALUES(PART_HISTORY_ID, CUSTOMER_ID, OPERATOR_ID, PART_NUMBER, DATE_DUE, PURCHASE_ORDER_NUMBER, QTY, OPERATIONS, DATE_RECEIVED , TO_DELETE);
END$$
